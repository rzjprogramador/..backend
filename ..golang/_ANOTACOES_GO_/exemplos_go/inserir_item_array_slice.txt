
// inserir item em array slice

var slice1 = []string{"foo", "bar"}
var str1 = "string1"
var str2 = "string2"
var str3 = "string3"

func addSliceMemory (s string) {
	slice1 = append(slice1, s)
}

func ExecuteSlice1() []string {
	addSliceMemory(str1)
	addSliceMemory(str2)
	addSliceMemory(str3)
	return slice1
}

ExecuteSlice1()
// output: [foo bar string1 string2 string3]
